// ── RECEIVE ORDERS — FIRESTORE LIVE ──────

// onSnapshot fires INSTANTLY when customer submits,

// then again whenever any field (e.g. status) changes.

// No polling. No refresh. Real time.


function handleNewOrder(docId, order){

  docIdMap[order.token] = docId;           // save docId for status updates

  if(orders.find(o => o.token === order.token)) return; // dedupe

  order.placedAt = order.placedAt || Date.now();

  orders.unshift(order);

  totalToday++;

  todayRevenue += order.grand;

  addLog('order', `🔔 NEW ORDER: ${order.token} from ${order.name} · ₹${order.grand}`, 'warn');

  renderKitchen();

  runOrderPipeline(order);

}


function handleOrderUpdate(docId, order){

  docIdMap[order.token] = docId;

  const existing = orders.find(o => o.token === order.token);

  if(existing){

    existing.status = order.status;

    existing.eta    = order.eta || existing.eta;

    renderKitchen();

  }

}


function startFirestoreListener(){

  // Listen to last 50 orders, newest first

  const q = query(collection(db, 'orders'), orderBy('placedAt', 'desc'), limit(50));

  const seenTokens = new Set();


  onSnapshot(q, snapshot => {

    snapshot.docChanges().forEach(change => {

      const docId = change.doc.id;

      const order = change.doc.data();

      if(change.type === 'added'){

        if(!seenTokens.has(order.token)){

          seenTokens.add(order.token);

          handleNewOrder(docId, order);

        }

      } else if(change.type === 'modified'){

        handleOrderUpdate(docId, order);

      }

    });

  }, err => {

    addLog('monitor', `Firestore error: ${err.message}`, 'err');

  });

}
